CREATE TABLE bazaVabanku (
    id        INTEGER        PRIMARY KEY AUTOINCREMENT,
    wartosc   INTEGER        NOT NULL,
    pytanie   NVARCHAR (200) NOT NULL,
    odpowiedz NVARCHAR (200) NOT NULL
                             DEFAULT "",
    kolumna   INTEGER        NOT NULL,
    wiersz    INTEGER        NOT NULL,
    UNIQUE (
        wiersz,
        kolumna
    )
);
